# Quick functions
mkcd() { mkdir -p "$1" && cd "$1"; }
extract() {
    if [ -f "$1" ]; then
        case "$1" in
            *.tar.bz2) tar xjf "$1" ;;
            *.tar.gz) tar xzf "$1" ;;
            *.bz2) bunzip2 "$1" ;;
            *.rar) unrar e "$1" ;;
            *.gz) gunzip "$1" ;;
            *.tar) tar xf "$1" ;;
            *.tbz2) tar xjf "$1" ;;
            *.tgz) tar xzf "$1" ;;
            *.zip) unzip "$1" ;;
            *.Z) uncompress "$1" ;;
            *.7z) 7z x "$1" ;;
            *) echo "'$1' cannot be extracted" ;;
        esac
    else
        echo "'$1' is not a valid file"
    fi
}

killport() {
  local pids=$(lsof -ti:$1)
  if [ -z "$pids" ]; then
    echo "No process running on port $1"
  else
    echo "$pids" | xargs kill -9
    echo "Killed process(es) on port $1"
  fi
}

mcp-clean() {
  local orphans=()
  while IFS= read -r line; do
    local pid=$(echo "$line" | awk '{print $1}')
    local ppid=$(echo "$line" | awk '{print $2}')
    # Walk up the process tree — if a launcher (uv/npm/node/python) has ppid=1,
    # it was reparented to launchd because its parent tool died = orphan
    local cur=$ppid
    local is_orphan=false
    while [[ $cur -gt 1 ]]; do
      local cur_ppid=$(ps -o ppid= -p $cur 2>/dev/null | tr -d ' ')
      [[ -z "$cur_ppid" ]] && { is_orphan=true; break; }  # parent died
      if [[ "$cur_ppid" -eq 1 ]]; then
        local cur_comm=$(ps -o comm= -p $cur 2>/dev/null)
        case "$cur_comm" in
          uv|npm|npx|node|python*) is_orphan=true ;;  # reparented launcher = orphan
        esac
        break
      fi
      cur=$cur_ppid
    done
    $is_orphan && orphans+=($pid)
  done < <(ps -eo pid,ppid,command | grep -E "[p]ostgres-mcp|[c]ontext7-mcp|[g]cloud-mcp|@azure/mcp|[w]hatsapp-mcp|[l]angsmith-mcp|[m]cp-grafana")
  if [[ ${#orphans[@]} -eq 0 ]]; then
    echo "No orphan MCP processes found"
  else
    echo "Killing ${#orphans[@]} orphan MCP processes..."
    kill -9 "${orphans[@]}" 2>/dev/null
    echo "Done"
  fi
}
